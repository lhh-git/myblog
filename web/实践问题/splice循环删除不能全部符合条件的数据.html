      • splice循环删除不能全部符合条件的数据

        • 解决方法：

# 问题原因:

因为splice每次删除都会改变原数组的长度，所以导致数组中的元素位置发生改变，从而导致只能删除部分数据

# 解决方法：

# 1. 倒序删除

let arr = [
  { id:1, name:'Tom', sex:0 }, 
  { id:2, name:'Jerry', sex:1 }, 
  { id:3, name:'Danny', sex:1 }, 
  { id:4, name:'Jenny', sex:0 }, 
  { id:6, name:'Lee', sex:0 }, 
  { id:5, name:'Bob', sex:1 }, 
  { id:7, name:'John', sex:0 }, 
  { id:8, name:'Marry', sex:1 }
];

for (let i = arr.length - 1; i >= 0; i--) {
  if (arr[i].sex === 1){
    arr.splice(i, 1);
  }
}

console.log(arr)
/* 
  [
    { id: 1, name: 'Tom', sex: 0 },
    { id: 4, name: 'Jenny', sex: 0 },
    { id: 6, name: 'Lee', sex: 0 },
    { id: 7, name: 'John', sex: 0 }
  ]
*/

# 2. 下标减1

let arr = [
  { id:1, name:'Tom', sex:0 }, 
  { id:2, name:'Jerry', sex:1 }, 
  { id:3, name:'Danny', sex:1 }, 
  { id:4, name:'Jenny', sex:0 }, 
  { id:6, name:'Lee', sex:0 }, 
  { id:5, name:'Bob', sex:1 }, 
  { id:7, name:'John', sex:0 }
];

for (let i = 0; i < arr.length; i++) {
	if (arr[i].sex === 1) {
    // 将使后面的元素依次前移，数组长度减1 
		arr.splice(i, 1)                         
    /*
      因为数组本身变了，长度变了，数组元素向前移了一步，所以循环也要往前移一步
      如果不减，将漏掉一个元素
    */
    i--                           
	}
}

console.log(arr)
/* 
  [
    { id: 1, name: 'Tom', sex: 0 },
    { id: 4, name: 'Jenny', sex: 0 },
    { id: 6, name: 'Lee', sex: 0 },
    { id: 7, name: 'John', sex: 0 }
  ]
*/
